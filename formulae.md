$$
\left.
  \begin{matrix*}[l]
    \text{1 coat}\\
    \text{10 lb. of tea}\\
    \text{40 lb. of coffee}\\
    \text{1 quarter of corn}\\
    \text{2 ounces of gold}\\
    \frac{1}{2}\text{ ton of iron}\\
    x\text{ commodity A etc.}
  \end{matrix*}
\right\}=\text{20 yards of linen}
$$
